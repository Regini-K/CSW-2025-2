            def calcular_total_com_desconto(itens):
    """
    Calcula o custo total de uma lista de itens, aplicando um desconto.
    """
    total_bruto = 0
    for preco in itens:
        total_bruto += preco

    desconto = 0
    if total_bruto > 100: # Define um limite para o desconto
        desconto = total_bruto * 0.10 # 10% de desconto
        print(f"Você recebeu um desconto de R${desconto:.2f}!")

    valor_final = total_bruto - desconto
    return valor_final
